ManyResult getMany(List<K> keys) {
  //Remove expired data O(n)
  _removeInteractionExpired();

  //Return map
  ManyResult<K, V> result = ManyResult();

  //Get keys
  // O(n)
  for (K key in keys) {
    MapEntry<K, V>? data = _get(key);

    if (data == null) {
      result.addNotFound(key);
      continue;
    }

    result.addFound(key, data.value);
  }

  return result;
}
